  <script>
    // ===== 질문만 여기 바꾸시면 됩니다 =====
    const QUESTIONS = [
      "올해 가장 기억에 남는 순간은?",
      "내년에 꼭 해보고 싶은 한 가지는?",
      "올해 가장 잘한 소비는?",
      "최근 가장 많이 들은 노래는?",
      "서로에게 고마웠던 점 한 가지씩 말하기",
      "지금 당장 여행 간다면 어디로 갈 건가요?",
      "올해 가장 크게 배운 교훈은?",
      "올해의 나를 한 단어로 표현한다면?"
    ];

    // ===== 상태 =====
    let deck = [];
    let total = 0;
    let used = 0;
    let started = false;
    let current = "";

    // Fisher–Yates shuffle
    function shuffle(arr){
      for(let i = arr.length - 1; i > 0; i--){
        const j = Math.floor(Math.random() * (i + 1));
        [arr[i], arr[j]] = [arr[j], arr[i]];
      }
      return arr;
    }

    function normalize(list){
      const cleaned = list.map(s => (typeof s === "string" ? s.trim() : "")).filter(Boolean);
      const seen = new Set();
      const uniq = [];
      for(const q of cleaned){
        if(!seen.has(q)){ seen.add(q); uniq.push(q); }
      }
      return uniq;
    }

    const elCard = document.getElementById("card");
    const elQ = document.getElementById("q");
    const elSub = document.getElementById("sub");
    const elProgress = document.getElementById("progress");

    const btnStart = document.getElementById("startBtn");
    const btnNext = document.getElementById("nextBtn");
    const btnReset = document.getElementById("resetBtn");

    function setCard(text, sub=""){
      elQ.textContent = text;
      elSub.textContent = sub;
    }

    function render(){
      const remaining = deck.length;
      if(!started){
        elProgress.textContent = "";
        btnNext.disabled = true;
        return;
      }
      elProgress.innerHTML = `Progress: <b>${used}</b>/<b>${total}</b> · Remaining: <b>${remaining}</b>`;
      btnNext.disabled = (remaining === 0);
    }

    function initDeck(){
      const q = normalize(QUESTIONS);
      deck = shuffle([...q]);
      total = deck.length;
      used = 0;
      current = "";
    }

    function flipNudge(){
      elCard.classList.add("flip");
      setTimeout(() => elCard.classList.remove("flip"), 220);
    }

    function start(){
      initDeck();
      started = true;
      if(total === 0){
        setCard("질문이 없습니다", "");
        render();
        return;
      }
      next(); // 시작하자마자 1개 보여주기
    }

    function next(){
      if(!started) return;
      if(deck.length === 0){
        setCard("끝!", "Reset을 누르면 다시 섞어서 시작합니다");
        render();
        return;
      }
      flipNudge();
      current = deck.pop(); // 중복 방지: 꺼내면 제거
      used += 1;
      setCard(current, "");
      if(deck.length === 0){
        // 마지막 질문이 나온 직후
        elSub.textContent = "마지막 질문입니다";
      }
      render();
    }

    function reset(){
      started = false;
      initDeck();
      setCard("Start를 눌러 질문을 시작하세요", "");
      render();
    }

    // 버튼
    btnStart.addEventListener("click", start);
    btnNext.addEventListener("click", next);
    btnReset.addEventListener("click", reset);

    // 카드 클릭/키보드
    elCard.addEventListener("click", () => {
      if(!started) start();
      else next();
    });
    elCard.addEventListener("keydown", (e) => {
      if(e.key === "Enter" || e.key === " "){
        e.preventDefault();
        if(!started) start();
        else next();
      }
    });

    // 초기 상태
    reset();
  </script>
